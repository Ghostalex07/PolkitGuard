Installation

# From source
go install github.com/Ghostalex07/PolkitGuard/cmd/scan@latest

# Docker
docker run ghostalex07/polkitguard:latest --path /etc/polkit-1

# Binary
curl -sL https://github.com/Ghostalex07/PolkitGuard/releases/latest/polkitguard-linux-amd64 -o /usr/local/bin/polkitguard
chmod +x /usr/local/bin/polkitguard

Usage Examples

# Basic scan
polkitguard scan --path /etc/polkit-1

# Scan with severity filter
polkitguard scan --severity high --format json

# Generate HTML report
polkitguard scan --path /etc/polkit-1 --format html --output report.html

# Watch mode
polkitguard scan --watch --interval 30s
